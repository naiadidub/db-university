SELECT * 
FROM `courses`
JOIN `degrees`
ON `courses`.`degree_id` = `degrees`.`id`
JOIN `departments`
ON `degrees`.`department_id` = `departments`.`id`
WHERE `degrees`.`level` = 'magistrale' 
AND `departments`.`name` = 'Dipartimento di Neuroscienze'

3: 
SELECT * 
FROM `courses`
JOIN `course_teacher`
ON `courses`.`id` = `course_teacher`.`course_id`
JOIN `teachers`
ON `course_teacher`.`teacher_id` = `teachers`.`id`
WHERE `teachers`.`name` = 'Fulvio' 
AND `teachers`.`surname` = 'Amato'

JOIN `degrees`

JOIN `departments`

SELECT `courses`.`degree_id` , 
`course_teacher`.`teacher_id`, 
`course_teacher`.`course_id` ,
`degrees`.`name` ,
`teachers`.`name`,
`teachers`.`surname`
FROM `degrees`
JOIN `courses`
ON `courses`.`degree_id` = `degrees`.`id`
JOIN `course_teacher`
ON `course_teacher`.`teacher_id` = `courses`.`id`
JOIN `teachers`
ON `course_teacher`.`course_id` = `teachers`.`id`
ORDER BY `courses`.`degree_id` ASC

6:
SELECT DISTINCT `teachers`.`name`, 
`teachers`.`surname`
FROM `teachers` 
JOIN `course_teacher`
ON `course_teacher`.`teacher_id` = `teachers`.`id`
JOIN `courses`
ON `courses`.`degree_id` = `course_teacher`.`course_id`
JOIN `degrees`
ON `degrees`.`department_id` = `courses`.`id`
JOIN `departments`
ON `departments`.`id` = `degrees`.`department_id`
WHERE `departments`.`id` = 'Dipartimento di Matematica'
